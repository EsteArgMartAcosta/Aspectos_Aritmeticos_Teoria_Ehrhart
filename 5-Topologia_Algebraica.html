import numpy as np, plotly.graph_objects as go
from IPython.display import HTML

R, r = 2.0, 1.0
epsilon = 0.05

def toro1(u, v, R=2, r=1):
    x = (R + r * np.cos(u)) * np.cos(v)
    y = (R + r * np.cos(u)) * np.sin(v)
    z = r * np.sin(u)
    return x, y, z

def toro2(u, v, R=2, r=1):
    x = r * np.sin(u)
    y = (R + r * np.cos(u)) * np.sin(v)
    z = (R + r * np.cos(u)) * np.cos(v)
    return x, y, z

n_points = 200
u = np.linspace(0, 2*np.pi, n_points)
v = np.linspace(0, 2*np.pi, n_points)
u, v = np.meshgrid(u, v)
u_flat, v_flat = u.flatten(), v.flatten()

x1, y1, z1 = toro1(u_flat, v_flat)
x2, y2, z2 = toro2(u_flat, v_flat)

pts1 = np.vstack((x1, y1, z1)).T
pts2 = np.vstack((x2, y2, z2)).T

tree2 = cKDTree(pts2)
idx = tree2.query_ball_point(pts1, r=epsilon)

intersection_pts = np.array([
    p1 for p1, match in zip(pts1, idx) if len(match) > 0
])

toro1_surface = go.Scatter3d(x=x1, y=y1, z=z1, mode='markers',
                             marker=dict(size=1, color='blue', opacity=0.05), name='Toro 1')

toro2_surface = go.Scatter3d(x=x2, y=y2, z=z2, mode='markers',
                             marker=dict(size=1, color='red', opacity=0.05), name='Toro 2')

intersection_plot = go.Scatter3d(x=intersection_pts[:,0],

                                 name='Intersecci√≥n')

layout = go.Layout(
    scene=dict(
        xaxis=dict(visible=False),
        yaxis=dict(visible=False),
        zaxis=dict(visible=False),
        aspectmode='data'
    ),
    title="Curva de intersecci√≥n aproximada de dos toros",
    showlegend=False
