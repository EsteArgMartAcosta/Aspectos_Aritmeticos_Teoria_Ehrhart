# Ejemplo de cálculo de homología simplicial en Python con Gudhi
import gudhi

simplex_tree = gudhi.SimplexTree()
simplex_tree.insert([0,1,2])
simplex_tree.insert([1,2,3])

print("Homología:")
print(simplex_tree.persistence())
